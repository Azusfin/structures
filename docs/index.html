BitSet

  • Set
  • Get
  • Flip
  • Clear

Init BitSet

 
import { BitSet } from "@azusfin/structures"

const bitsAmount = 64
const bitSet = new BitSet(bitsAmount)

Set Bit

 
const index = 31
bitSet.set(index)

Get Bit

 
bitSet.get(31) // true
bitSet.get(4) // false

Flip BIt

 
bitSet.flip(31)
bitSet.flip(4)

bitSet.get(31) // false
bitSet.get(4) // true

Clear Bit

 
bitSet.clear(31)
bitSet.clear(4)

bitSet.get(31) // false
bitSet.get(4) // false
